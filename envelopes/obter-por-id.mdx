---
title: Obter Envelope por ID
description: Obtém detalhes completos de um envelope específico.
openapi: 'GET /api/envelopes/{envelopeId}'
---

Este endpoint retorna todos os detalhes de um envelope específico, incluindo campos, destinatários, histórico e configurações.

<RequestExample dropdown>

```bash cURL
curl -X GET "https://api.formsign.com.br/api/envelopes/env_abc123456789" \
  -H "Authorization: Bearer <TOKEN>"
```

```javascript JavaScript
const envelopeId = 'env_abc123456789';
const response = await fetch(`https://api.formsign.com.br/api/envelopes/${envelopeId}`, {
  headers: {
    'Authorization': 'Bearer <TOKEN>'
  }
});

if (response.ok) {
  const envelope = await response.json();
  console.log('=== DETALHES DO ENVELOPE ===');
  console.log(`Assunto: ${envelope.subject}`);
  console.log(`Status: ${envelope.status}`);
  console.log(`Criado em: ${envelope.creationDate}`);
  console.log(`Atualizado em: ${envelope.lastUpdateDate}`);
  
  console.log('\n=== DESTINATÁRIOS ===');
  envelope.recipients.forEach((recipient, index) => {
    console.log(`${index + 1}. ${recipient.name} (${recipient.email})`);
    console.log(`   Status: ${recipient.status} | Papel: ${recipient.role}`);
  });
  
  console.log('\n=== CAMPOS DO DOCUMENTO ===');
  envelope.fields.forEach(field => {
    console.log(`- ${field.name || field.identifier}: ${field.type} (${field.required ? 'obrigatório' : 'opcional'})`);
  });
  
  if (envelope.observers && envelope.observers.length > 0) {
    console.log('\n=== OBSERVADORES ===');
    envelope.observers.forEach(observer => {
      console.log(`- ${observer.name} (${observer.email})`);
    });
  }
} else {
  console.error('Erro ao buscar envelope:', response.status);
}
```

```python Python
import requests
from datetime import datetime

envelope_id = 'env_abc123456789'
response = requests.get(
    f'https://api.formsign.com.br/api/envelopes/{envelope_id}',
    headers={'Authorization': 'Bearer <TOKEN>'}
)

if response.status_code == 200:
    envelope = response.json()
    
    print("=== DETALHES DO ENVELOPE ===")
    print(f"Assunto: {envelope['subject']}")
    print(f"Status: {envelope['status']}")
    print(f"Tipo: {envelope['submissionType']}")
    print(f"Criado em: {envelope['creationDate']}")
    print(f"Atualizado em: {envelope['lastUpdateDate']}")
    
    if envelope.get('sentDate'):
        print(f"Enviado em: {envelope['sentDate']}")
    
    if envelope.get('completedDate'):
        print(f"Completado em: {envelope['completedDate']}")
    
    if envelope.get('expirationDate'):
        print(f"Expira em: {envelope['expirationDate']}")
    
    print(f"\n=== REMETENTE ===")
    sender = envelope['sender']
    print(f"Nome: {sender['userName']}")
    print(f"Email: {sender['email']}")
    
    print(f"\n=== DESTINATÁRIOS ({len(envelope['recipients'])}) ===")
    for i, recipient in enumerate(envelope['recipients'], 1):
        print(f"{i}. {recipient['name']} ({recipient['email']})")
        print(f"   Status: {recipient['status']} | Papel: {recipient['role']}")
        print(f"   Tipo de Assinatura: {recipient['signatureType']}")
        if recipient.get('cpf'):
            print(f"   CPF/CNPJ: {recipient['cpf']}")
        if recipient.get('completedDate'):
            print(f"   Assinado em: {recipient['completedDate']}")
        print()
    
    print(f"=== CAMPOS DO DOCUMENTO ({len(envelope['fields'])}) ===")
    for field in envelope['fields']:
        field_name = field.get('name', field['identifier'])
        required_text = "obrigatório" if field['required'] else "opcional"
        print(f"- {field_name}: {field['type']} ({required_text})")
        print(f"  Posição: x={field['x']}%, y={field['y']}%, w={field['w']}%, h={field['h']}%")
        if field.get('pageNumber'):
            print(f"  Página: {field['pageNumber']}")
        if field.get('recipientRole'):
            print(f"  Responsável: {field['recipientRole']}")
        print()
    
    if envelope.get('observers') and len(envelope['observers']) > 0:
        print(f"=== OBSERVADORES ({len(envelope['observers'])}) ===")
        for observer in envelope['observers']:
            print(f"- {observer['name']} ({observer['email']})")
    
    print("=== OPÇÕES AVANÇADAS ===")
    adv_opts = envelope['advancedOptions']
    print(f"Lembretes automáticos: {'Sim' if adv_opts['autoReminders'] else 'Não'}")
    print(f"Atraso inicial: {adv_opts['reminderDelay']} dias")
    print(f"Frequência: {adv_opts['reminderFrequency']} dias")
    print(f"Expiração: {adv_opts['expirationDays']} dias")
    print(f"Aviso de expiração: {adv_opts['expirationWarning']} dias")
    
else:
    print(f"Erro ao buscar envelope: {response.status_code}")
    print(response.json())
```
